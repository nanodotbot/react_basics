Routing setup and route definitions.

import { BrowserRouter as Router, Routes, Route } from 'react-router-dom';
import './App.css';
import Layout from './components/Layout';
import Home from './pages/Home';
import About from './pages/About';

function App() {
    return (
        <Router>
            <Routes>
                {/* Layout Component Wrapping Pages */}
                <Route path="/" element={<Layout />}>
                    <Route index element={<Home />} /> {/* Default route */}
                    <Route path="about" element={<About />} />
                </Route>
            </Routes>
        </Router>
    );
}

export default App;

Folder structure

The most common approach is to store components inside a src/components/ directory. You can create subfolders for specific types of components, but for now, simple components like Header and Footer can be placed directly under the components folder.

src/
    components/
        Layout.js
        Header.js
        Footer.js
    pages/
        Home.js
        About.js
    styles/
        Header.css
        Footer.css

Component File Naming

Use PascalCase (capitalizing the first letter of each word) for component file names to follow convention.

src/App.js

React Router tries to match the userâ€™s URL against the defined paths (e.g., /, /about). | If no route matches, it falls back to the path="*" route, rendering the NotFound component.

import { BrowserRouter as Router, Routes, Route } from 'react-router-dom';
import './App.css';
import Layout from './components/Layout';
import Home from './pages/Home';
import About from './pages/About';
import NotFound from './pages/NotFound';

function App() {
    return (
        <Router>
            <Routes>
                {/* Layout Component Wrapping Pages */}
                <Route path="/" element={<Layout />}>
                    <Route index element={<Home />} /> {/* Default route */}
                    <Route path="about" element={<About />} />
                    {/* The 404 Route */}
                    <Route path="*" element={<NotFound />} />
                </Route>
            </Routes>
        </Router>
    );
}

export default App;

You can decide where to place the 404 route. If you put it inside the <Layout> route, the 404 page will still render with your layout (e.g., header and footer). If you want a standalone 404 page without the layout, you could define a path="*" route outside of the <Layout> route. | Make sure this catch-all route is at the end of your <Routes> block (although in v6, the order doesnâ€™t matter as much, because the matching logic is different than older versions). | If you have nested routes, you can add an additional path="*" route at each level if needed.

Using document.title Manually

// src/pages/About.js
import React from 'react';

const About = () => {
    React.useEffect(() => {
        document.title = 'About | My React App';
    }, []);

    // Change meta description
    const metaDescription = document.querySelector('meta[name="description"]');
    if (metaDescription) {
        metaDescription.setAttribute('content', 'This is the About page description.');
    }

    return (
        <div>
            <h2>About This App</h2>
            <p>This is the About page content.</p>
        </div>
    );
};

export default About;

Which Method to Choose?

React Helmet (or react-helmet-async) is preferred if: You have multiple meta tags, including description, keywords, or Open Graph tags for social sharing. | You want a more declarative approach, placing head elements near each pageâ€™s content.

Manual document.title is simpler if: You only need to set the title and donâ€™t frequently handle other meta tags. | You prefer a minimal approach without extra libraries.

// src/pages/UsersFetch.js
import React from 'react';

const UsersFetch = () => {
    const [users, setUsers] = React.useState([]);
    const [loading, setLoading] = React.useState(false);
    const [error, setError] = React.useState(null);

    React.useEffect(() => {
        setLoading(true);
        fetch('https://jsonplaceholder.typicode.com/users') // Replace with your API endpoint
            .then(response => {
                if (!response.ok) {
                    throw new Error(`HTTP error! Status: ${response.status}`);
                }
                return response.json();
            })
            .then(data => {
                setUsers(data);
                setError(null);
            })
            .catch(err => {
                setError(err.message);
            })
            .finally(() => {
                setLoading(false);
            });
    }, []);

    if (loading) return <p>Loading users...</p>;
    if (error) return <p style={{ color: 'red' }}>Error: {error}</p>;

    return (
        <div>
            <h2>Users (Fetch)</h2>
            <ul>
                {users.map(user => (
                    <li key={user.id}>
                        {user.name} â€“ {user.email}
                    </li>
                ))}
            </ul>
        </div>
    );
};

export default UsersFetch;

// src/pages/UsersAxios.js
import React from 'react';
import axios from 'axios';

const UsersAxios = () => {
    const [users, setUsers] = React.useState([]);
    const [loading, setLoading] = React.useState(false);
    const [error, setError] = React.useState(null);

    React.useEffect(() => {
        const fetchUsers = async () => {
            try {
                setLoading(true);
                const response = await axios.get('https://jsonplaceholder.typicode.com/users');
                // Replace with your API endpoint
                setUsers(response.data);
                setError(null);
            } catch (err) {
                setError(err.message);
            } finally {
                setLoading(false);
            }
        };

        fetchUsers();
    }, []);

    if (loading) return <p>Loading users...</p>;
    if (error) return <p style={{ color: 'red' }}>Error: {error}</p>;

    return (
        <div>
            <h2>Users (Axios)</h2>
            <ul>
                {users.map(user => (
                    <li key={user.id}>
                        {user.name} â€“ {user.email}
                    </li>
                ))}
            </ul>
        </div>
    );
};

export default UsersAxios;

Features of Axios

Automatic JSON Transform: Axios automatically parses JSON responses, so you donâ€™t have to manually call .json(). | Better Error Handling: Errors are consolidated, and you donâ€™t have to manually check response.ok. | Interceptors & More: Axios provides powerful features like interceptors (for token refresh, custom headers, logging, etc.).

Create

import React from 'react';

const CreateUser = () => {
    const [name, setName] = React.useState('');
    const [email, setEmail] = React.useState('');
    const [loading, setLoading] = React.useState(false);
    const [error, setError] = React.useState(null);
    const [success, setSuccess] = React.useState(null);

    const handleSubmit = (e) => {
        e.preventDefault();

        // Reset previous states
        setError(null);
        setSuccess(null);
        setLoading(true);

        // Perform the POST request (using a public test API endpoint, or your own)
        fetch('https://jsonplaceholder.typicode.com/users', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({
                // The data to send
                name,
                email,
            }),
        })
            .then((response) => {
                if (!response.ok) {
                    // If server returns an error status
                    throw new Error(`HTTP error! Status: ${response.status}`);
                }
                return response.json();
            })
            .then((data) => {
                // Clear form fields
                setName('');
                setEmail('');
                setSuccess('User created successfully!');
            })
            .catch((err) => {
                setError(err.message);
            })
            .finally(() => {
                setLoading(false);
            });
    };

    return (
        <div style={{ padding: '20px' }}>
            <h2>Create a New User</h2>

            {/* Feedback Messages */}
            {loading && <p>Creating user...</p>}
            {error && <p style={{ color: 'red' }}>{error}</p>}
            {success && <p style={{ color: 'green' }}>{success}</p>}

            {/* The Form */}
            <form onSubmit={handleSubmit} style={{ maxWidth: '300px' }}>
                <div style={{ marginBottom: '10px' }}>
                    <label>
                        Name:
                        <input
                            type="text"
                            value={name}
                            onChange={(e) => setName(e.target.value)}
                            required
                            style={{ display: 'block', width: '100%' }}
                        />
                    </label>
                </div>

                <div style={{ marginBottom: '10px' }}>
                    <label>
                        Email:
                        <input
                            type="email"
                            value={email}
                            onChange={(e) => setEmail(e.target.value)}
                            required
                            style={{ display: 'block', width: '100%' }}
                        />
                    </label>
                </div>

                <button type="submit" disabled={loading}>
                    Create User
                </button>
            </form>
        </div>
    );
};

export default CreateUser;
